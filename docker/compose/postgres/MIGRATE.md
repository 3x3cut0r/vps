# create databases
psql -U postgres -d postgres <<'SQL'
CREATE ROLE openwebui LOGIN PASSWORD 'OPENWEBUI_PASSWORD';
CREATE DATABASE openwebui OWNER openwebui;
REVOKE ALL ON DATABASE openwebui FROM PUBLIC;
GRANT CONNECT ON DATABASE openwebui TO openwebui;

CREATE ROLE litellm   LOGIN PASSWORD 'LITELLM_PASSWORD';
CREATE DATABASE litellm   OWNER litellm;
REVOKE ALL ON DATABASE litellm   FROM PUBLIC;
GRANT CONNECT ON DATABASE litellm   TO litellm;

CREATE ROLE n8n   LOGIN PASSWORD 'N8N_PASSWORD';
CREATE DATABASE n8n   OWNER n8n;
REVOKE ALL ON DATABASE n8n   FROM PUBLIC;
GRANT CONNECT ON DATABASE n8n   TO n8n;
SQL
